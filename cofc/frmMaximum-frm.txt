Option Explicit ' require type declarations

Private Sub cmdRun_Click()
    Dim list(10) As Integer ' a list of 10 numbers
    Dim i As Integer ' index, identifies one element of list
    Dim max As Integer ' the largest number in the list
    
    ' Initialize form printing
    frmMaximum.Cls ' clear screen of printing from last time
    frmMaximum.CurrentY = 220 ' looks good with 18 pt text
    For i = 1 To 3
        frmMaximum.Print ' leave some space
    Next i
    
    ' Fill the list with random numbers and print
    max = 0 ' initialize maximum
    For i = 1 To 10
        list(i) = Int(100 * Rnd())
        If list(i) > max Then max = list(i)
        frmMaximum.Print "   i "; i; "  list(i) "; list(i)
    Next i
    
    ' Report the maximum
    frmMaximum.Print ' leave some space
    frmMaximum.Print "   The largest number is "; max
End Sub

Private Sub Form_Load()
    Randomize ' otherwise rnd always returns same sequence
End Sub
